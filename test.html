    function sendPostMessage(data) {

      window.webkit.messageHandlers.postMessageHandler.postMessage(data);

    }


    function sendFirstButtonPost() {

      const address = 'random_address';

      const twitterId = 'random_twitterId';

      const event = 'random_event';


      const data = {

        address: address,

        twitterId: twitterId,

        event: event

      };


      sendPostMessage(data);

    }


    function sendSecondButtonPost() {

      const message = 'random_message';


      const data = {

        message: message

      };


      sendPostMessage(data);

    }


    function sendThirdButtonPost() {

      const url = 'https://www.google.com';


      const data = {

        url: url

      };


      sendPostMessage(data);

    }
